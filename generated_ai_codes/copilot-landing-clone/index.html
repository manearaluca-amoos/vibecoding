function calculateDaysBetweenDates(begin: Date, end: Date): number {
    const diff = end.getTime() - begin.getTime();
    return Math.floor(diff / (1000 * 60 * 60 * 24));
}

// Copilot suggestion
function formatDateRange(begin: Date, end: Date): string {
    const days = calculateDaysBetweenDates(begin, end);
    return `${days} day${days !== 1 ? 's' : ''} between ${begin.toLocaleDateString()} and ${end.toLocaleDateString()}`;
}
